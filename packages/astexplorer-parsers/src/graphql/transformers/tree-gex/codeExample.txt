import * as w from "tree-gex";

// Find all Query/Mutation actions and their @isId-annotated arguments.

const argWithIsIdDirective = {
  kind: "InputValueDefinition",
  name: { kind: "Name", value: w.group(w.string(), "arg") },
  directives: [{ kind: "Directive", name: { kind: "Name", value: "isId" } }],
};

const actionDefinition = {
  kind: "FieldDefinition",
  name: { kind: "Name", value: w.group(w.string(), "action") },
  arguments: w.arrayZeroOrOne(argWithIsIdDirective),
};

export default w.accumWalkMatch(ast, {
  kind: "ObjectTypeDefinition",
  name: { value: w.regex(/Query|Mutation/) },
  fields: w.arrayFor(w.group(actionDefinition, "actions")),
});
