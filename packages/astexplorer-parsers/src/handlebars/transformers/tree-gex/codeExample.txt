import { accumWalkMatch, string, group } from "tree-gex";

// Find all MustacheStatement helpers in the template.
// `ast` is the Glimmer AST of the Handlebars code.
export default accumWalkMatch(ast, {
  type: "MustacheStatement",
  path: {
    type: "PathExpression",
    original: group(string(), "helper"),
  },
});
