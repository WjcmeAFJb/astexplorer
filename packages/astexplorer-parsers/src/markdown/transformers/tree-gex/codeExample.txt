const w = require("tree-gex");

// Match all heading nodes and capture their depth and text
module.exports = {
  pattern: {
    type: "heading",
    depth: w.group(w.any(), "depth"),
    children: w.arrayFor(
      w.group({ type: "text", value: w.group(w.string(), "text") }, "children"),
    ),
  },
};
