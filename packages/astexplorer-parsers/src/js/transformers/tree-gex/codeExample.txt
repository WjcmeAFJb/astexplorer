import { walkReplace, transform, string, group } from "tree-gex";

// Replace all function names by reversing them.
// `ast` is the parsed AST of the code in the left pane.
// The return value is rendered back to source code via recast.
export default walkReplace(ast, {
  type: "FunctionDeclaration",
  id: {
    type: "Identifier",
    name: transform(string(), (name) => ({
      type: "Identifier",
      name: name.split("").reverse().join(""),
    })),
  },
});
