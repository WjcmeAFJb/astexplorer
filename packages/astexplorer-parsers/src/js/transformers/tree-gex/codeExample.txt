import * as w from "tree-gex";

// Reverse all function names in the code.
// The result is printed back to source via recast.
export default w.walkReplace(ast, {
  type: "FunctionDeclaration",
  id: {
    type: "Identifier",
    name: w.transform(w.string(), (name) =>
      name.split("").reverse().join("")
    ),
  },
});
