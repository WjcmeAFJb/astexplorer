import { walkReplace, transform, string } from "tree-gex";

// Reverse all function names in the code.
// `ast` is the parsed AST. The result is printed back to source via recast.
export default walkReplace(ast, {
  type: "FunctionDeclaration",
  id: {
    type: "Identifier",
    name: transform(string(), (name) =>
      name.split("").reverse().join("")
    ),
  },
});
